Agent OS wraps existing agent frameworks with minimal code changes. Your agents gain kernel-level governance without rewriting your application.

ðŸ¦œ LangChain

Wrap LangChain agents and chains with policy enforcement.

from langchain.agents import AgentExecutor, create_openai_functions_agent
from langchain_openai import ChatOpenAI
from langchain.tools import tool
from agent_os.integrations import LangChainKernel

# Your existing LangChain setup
llm = ChatOpenAI(model="gpt-4")

@tool
def search_database(query: str) -> str:
    """Search the customer database."""
    return f"Results for: {query}"

tools = [search_database]
agent = create_openai_functions_agent(llm, tools, prompt)
agent_executor = AgentExecutor(agent=agent, tools=tools)

# Wrap with Agent OS - ONE LINE
kernel = LangChainKernel(policy="strict")
governed_agent = kernel.wrap(agent_executor)

# Use exactly as before - now governed
result = governed_agent.invoke({"input": "Find all premium customers"})
# Policies enforced on every tool call

Features

  • Intercepts all tool calls for policy checks
  • Logs agent reasoning chain
  • Supports streaming with policy enforcement
  • Works with LCEL chains

Advanced: Chain Integration

from langchain_core.runnables import RunnablePassthrough
from agent_os.integrations import LangChainKernel

# Wrap individual chain components
kernel = LangChainKernel(policy="read_only")

chain = (
    {"context": kernel.wrap(retriever), "question": RunnablePassthrough()}
    | prompt
    | kernel.wrap(llm)  # LLM calls governed too
    | output_parser
)

result = chain.invoke("What are our Q3 results?")

ðŸ‘¥ CrewAI

Govern entire CrewAI crews with multi-agent policy enforcement.

from crewai import Agent, Task, Crew
from agent_os.integrations import CrewAIKernel

# Your existing CrewAI setup
researcher = Agent(
    role="Research Analyst",
    goal="Find market trends",
    backstory="Expert market researcher"
)

writer = Agent(
    role="Content Writer",
    goal="Create reports",
    backstory="Technical writer"
)

research_task = Task(
    description="Research AI market trends",
    agent=researcher
)

write_task = Task(
    description="Write summary report",
    agent=writer
)

crew = Crew(agents=[researcher, writer], tasks=[research_task, write_task])

# Wrap with Agent OS
kernel = CrewAIKernel(policy="strict")
governed_crew = kernel.wrap(crew)

# Execute with governance
result = governed_crew.kickoff()
# All agent actions governed, inter-agent communication logged

Multi-Agent Policies

# Different policies per agent role
kernel = CrewAIKernel(
    default_policy="strict",
    agent_policies={
        "Research Analyst": "read_only",      # Can only read
        "Content Writer": "no_network",       # No external calls
        "Admin Agent": "permissive"           # Full access
    }
)

Wrap Microsoft AutoGen agents and group chats.

from autogen import AssistantAgent, UserProxyAgent
from agent_os.integrations import AutoGenKernel

# Your existing AutoGen setup
assistant = AssistantAgent(
    name="assistant",
    llm_config={"model": "gpt-4"}
)

    name="user",
    human_input_mode="NEVER"
)

governed_proxy = kernel.wrap(user_proxy)

# Chat with governance
governed_proxy.initiate_chat(
    governed_assistant,
    message="Analyze our sales data"
)

ðŸ”® OpenAI Assistants

Govern OpenAI Assistants API with tool call interception.

from openai import OpenAI
from agent_os.integrations import OpenAIKernel

client = OpenAI()

# Create an assistant
assistant = client.beta.assistants.create(
    name="Data Analyst",
    instructions="You analyze data and create reports.",
    model="gpt-4-turbo",
    tools=[{"type": "code_interpreter"}]
)

# Wrap with Agent OS
kernel = OpenAIKernel(policy="strict")
governed = kernel.wrap_assistant(assistant, client)

# Create thread and run - governed
thread = governed.threads.create()
governed.threads.messages.create(
    thread_id=thread.id,
    role="user",
    content="Analyze the attached CSV"
)

run = governed.threads.runs.create(
    thread_id=thread.id,
    assistant_id=assistant.id
)
# Code interpreter runs governed - file access controlled

Function Calling Governance

# Tool calls are intercepted and governed
tools = [
    {
        "type": "function",
        "function": {
            "name": "query_database",
            "description": "Query the database",
            "parameters": {...}
        }
    }
]

# When the assistant calls query_database:
# 1. Kernel intercepts the call
# 2. Checks against policy
# 3. Blocks if policy violation
# 4. Logs all calls to flight recorder

ðŸ’  Semantic Kernel

Integrate with Microsoft Semantic Kernel for enterprise AI.

import semantic_kernel as sk
from semantic_kernel.connectors.ai.open_ai import OpenAIChatCompletion
from agent_os.integrations import SemanticKernelIntegration

# Your existing Semantic Kernel setup
kernel = sk.Kernel()
kernel.add_chat_service(
    "chat-gpt",
    OpenAIChatCompletion("gpt-4", api_key="...")
)

# Add Agent OS governance
agent_os = SemanticKernelIntegration(policy="strict")
agent_os.register(kernel)

# All SK functions now governed
@kernel.register_native_function
def search_documents(query: str) -> str:
    # This function call is governed by Agent OS
    return search(query)

result = await kernel.run_async(
    kernel.skills.get_function("search_documents"),
    input_str="Find Q3 reports"
)

ðŸ¦™ LlamaIndex

Govern LlamaIndex queries and agents.

from llama_index import VectorStoreIndex, SimpleDirectoryReader
from llama_index.agent import OpenAIAgent
from agent_os.integrations import LlamaIndexKernel

# Your existing LlamaIndex setup
documents = SimpleDirectoryReader("data/").load_data()
index = VectorStoreIndex.from_documents(documents)
query_engine = index.as_query_engine()

# Wrap with Agent OS
kernel = LlamaIndexKernel(policy="read_only")
governed_engine = kernel.wrap(query_engine)

# Queries governed
response = governed_engine.query("What are our key metrics?")

# Agent mode
agent = OpenAIAgent.from_tools([tool1, tool2])
governed_agent = kernel.wrap_agent(agent)

response = governed_agent.chat("Summarize documents")
